anonymus

Generates random, unidentifiable names for any purpose

Installation

In a browser:

<script src="https://cdn.jsdelivr.net/npm/anonymus@1.3.1/index.min.js"></script>

npm

npm i anonymus --save

Usage

npm

const anonymus = require("anonymus");

console.log(anonymus.create());
//=> "Orange panda"

console.log(anonymus.create());
//=> "Green hawk"

console.log(anonymus.create());
//=> "Blue sheep"

console.log(anonymus.create(5));
//=> [ 'Red hummingbird',
//     'White frog',
//     'Pink jellyfish',
//     'Sapphire kangaroo',
//     'Black dog' ]

console.log(anonymus.create(anonymus.sizeMax));
//=> Creates all unique possibilities.
